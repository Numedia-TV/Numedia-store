    const tbody = document.querySelector('#products tbody');


    function formatUSD(v){

      return '$' + Number(v).toFixed(2);

    }


    function render(rate){

      tbody.innerHTML = '';

      products.forEach((p, idx)=>{

        const usd = (p.dzd * rate);

        const tr = document.createElement('tr');

        tr.innerHTML = `

          <td>${p.code}</td>

          <td data-dzd>${p.dzd}</td>

          <td data-usd>${formatUSD(usd)}</td>

          <td><input type="number" step="0.01" value="${usd.toFixed(2)}" data-idx="${idx}" /></td>

          <td><button class="btn btn-telegram" data-idx="${idx}">Payer via Telegram</button></td>

        `;

        tbody.appendChild(tr);

      });


      // attach listeners

      tbody.querySelectorAll('input[type=number]').forEach(inp=>{

        inp.addEventListener('change', (e)=>{

          const i = +e.target.dataset.idx; products[i].customUsd = Number(e.target.value);

        });

      });


      tbody.querySelectorAll('.btn-telegram').forEach(b=>{

        b.addEventListener('click', ()=>{

          const i = +b.dataset.idx;

          const p = products[i];

          const price = p.customUsd ? p.customUsd : (p.dzd * Number(document.getElementById('rate').value));

          // Open Telegram channel (edit link below to your actual channel)

          const tg = 'https://t.me/YourChannel';

          // We add product info in message param (user can see product name in channel or pinned message)

          window.open(tg, '_blank');

        });

      });

    }


    document.getElementById('convert').addEventListener('click', ()=>{

      const rate = Number(document.getElementById('rate').value) || 0.0077;

      render(rate);

      localStorage.setItem('numedia_rate', rate);

    });


    document.getElementById('exportCsv').addEventListener('click', ()=>{

      const rate = Number(document.getElementById('rate').value) || 0.0077;

      let csv = 'Code,Price_DZD,Price_USD\n';

      products.forEach(p=>{

        const usd = p.customUsd ? p.customUsd : (p.dzd * rate);

        csv += `"${p.code}",${p.dzd},${usd.toFixed(2)}\n`;

      });

      const blob = new Blob([csv],{type:'text/csv'});

      const url = URL.createObjectURL(blob);

      const a = document.createElement('a'); a.href=url; a.download='numedia_products.csv'; a.click(); URL.revokeObjectURL(url);

    });


    document.getElementById('reset').addEventListener('click', ()=>{

      localStorage.removeItem('numedia_rate');

      document.getElementById('rate').value='0.0077';

      render(0.0077);

    });


    // init

    const savedRate = localStorage.getItem('numedia_rate');

    if(savedRate) document.getElementById('rate').value = savedRate;

    render(Number(document.getElementById('rate').value));
